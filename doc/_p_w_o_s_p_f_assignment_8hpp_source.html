17constexpr Size MaxCost = 20;
18constexpr Size calc_cost_lcm(size_t a) {
19 return a == 1 ? 1 : std::lcm(a, calc_cost_lcm(a - 1));
20}
21constexpr Size CostLCM = calc_cost_lcm(MaxCost);
